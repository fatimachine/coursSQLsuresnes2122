Les fonctions prédéfinies

1-Quelques fonctions

Fonctions prédéfinies Description
SELECT CURDATE() Affiche la date quand on abesoin de la date du jour dans la requête
SELECT CURDATE() + 0 Affiche la date courante sans tirets
SELECT CURTIME() Affiche l'heure
SELECT NOW() Date et heure
SELECT DATE_ADD('2021-12-06',INTERVAL 31 DAY/MONTH/YEAR) Ajoute xx jour/mois/année à la date de départ
SELECT DATE_FORMAT(NOW(),'%d/%m%Y - %H:%i:%s'); Autre exemple : SELECT date_rendu, DATE_FORMAT(date_rendu, '%d/%m/%Y') FROM emprunt Autre exemple : SELECT date_rendu, DATE_FORMAT(date_rendu, '%d/%m/%Y')AS 'date FR' FROM emprunt; Change le format de la date du jour au format FR
SELECT CONCAT_WS(' - ','a','b','c') WS with separator utile pour concaténer un code postal avec une ville
SELECT TRIM(' coucou ') Supprime les espaces en début et en fin de chaîne
